# ESP32 Code Fix - Servo chỉ mở sau khi camera nhận diện

## Vấn đề hiện tại:
ESP32 đang mở servo NGAY SAU khi RFID scan → SAI!

## Logic đúng:
ESP32 chỉ mở servo KHI NHẬN LỆNH từ backend (sau khi camera nhận diện xong)

## Code ESP32 cần sửa:

### TRƯỚC (SAI):
```cpp
void loop() {
  // Đọc RFID
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    String uid = getUID();
    
    // Publish MQTT
    publishRFIDScan(uid, "IN");
    
    // ❌ SAI: Mở servo ngay lập tức
    openServo();  // <-- XÓA DÒNG NÀY!
  }
}
```

### SAU (ĐÚNG):
```cpp
void loop() {
  // Đọc RFID
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    String uid = getUID();
    
    // Chỉ publish MQTT, KHÔNG mở servo
    publishRFIDScan(uid, "IN");
    
    // ✅ ĐÚNG: Không mở servo ở đây
    // Servo sẽ được mở khi nhận lệnh từ callback()
  }
  
  client.loop();  // Đảm bảo MQTT callback được gọi
}

// Callback nhận lệnh từ backend
void callback(char* topic, byte* payload, unsigned int length) {
  if (strcmp(topic, "smart_parking/servo/command") == 0) {
    StaticJsonDocument<256> doc;
    deserializeJson(doc, payload, length);
    
    String action = doc["action"];
    String gate = doc["gate"];
    String license_plate = doc["license_plate"];
    
    // ✅ ĐÚNG: Chỉ mở servo khi nhận lệnh từ backend
    if (action == "open_gate" && gate == "IN") {
      Serial.print("Opening gate for vehicle: ");
      Serial.println(license_plate);
      
      openServo();  // <-- MỞ SERVO Ở ĐÂY!
      delay(5000);
      closeServo();
    }
  }
}
```

## Code đầy đủ (tham khảo):

```cpp
#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <MFRC522.h>
#include <ESP32Servo.h>

// MQTT
WiFiClient espClient;
PubSubClient client(espClient);

// RFID
MFRC522 rfid(SS_PIN, RST_PIN);

// Servo
Servo servoIN;

void setup() {
  Serial.begin(115200);
  
  // Setup WiFi
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
  
  // Setup MQTT
  client.setServer(MQTT_BROKER, MQTT_PORT);
  client.setCallback(callback);  // ← QUAN TRỌNG!
  
  // Setup RFID
  SPI.begin();
  rfid.PCD_Init();
  
  // Setup Servo
  servoIN.attach(SERVO_PIN);
  servoIN.write(0);  // Đóng cửa ban đầu
  
  Serial.println("System ready");
}

void loop() {
  // Reconnect MQTT nếu mất kết nối
  if (!client.connected()) {
    reconnect();
  }
  client.loop();  // ← QUAN TRỌNG: Để callback được gọi
  
  // Đọc RFID
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    String uid = getUID();
    Serial.print("RFID detected: ");
    Serial.println(uid);
    
    // CHỈ publish MQTT, KHÔNG mở servo
    publishRFIDScan(uid, "IN");
    
    rfid.PICC_HaltA();
    rfid.PCD_StopCrypto1();
  }
  
  delay(100);
}

// Callback nhận lệnh từ backend
void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("]: ");
  
  // Parse JSON
  StaticJsonDocument<256> doc;
  DeserializationError error = deserializeJson(doc, payload, length);
  
  if (error) {
    Serial.println("JSON parse failed");
    return;
  }
  
  // In ra để debug
  serializeJsonPretty(doc, Serial);
  Serial.println();
  
  // Xử lý lệnh servo
  if (strcmp(topic, "smart_parking/servo/command") == 0) {
    String action = doc["action"];
    String gate = doc["gate"];
    String license_plate = doc["license_plate"];
    
    if (action == "open_gate" && gate == "IN") {
      Serial.println("=================================");
      Serial.print("✅ Opening gate for vehicle: ");
      Serial.println(license_plate);
      Serial.println("=================================");
      
      // MỞ SERVO
      servoIN.write(90);  // Góc mở (điều chỉnh theo servo của bạn)
      delay(5000);        // Giữ mở 5 giây
      servoIN.write(0);   // Đóng lại
      
      Serial.println("Gate closed");
    }
  }
}

void publishRFIDScan(String uid, String gate) {
  StaticJsonDocument<200> doc;
  doc["event"] = "rfid_scan";
  doc["gate"] = gate;
  doc["uid"] = uid;
  
  char buffer[200];
  serializeJson(doc, buffer);
  
  client.publish("smart_parking/rfid", buffer);
  Serial.print("Published RFID scan: ");
  Serial.println(buffer);
}

String getUID() {
  String uid = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    uid += String(rfid.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();
  return uid;
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    if (client.connect("ESP32_IN_Gate")) {
      Serial.println("connected");
      
      // Subscribe topic servo command
      client.subscribe("smart_parking/servo/command");
      Serial.println("Subscribed to: smart_parking/servo/command");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      delay(5000);
    }
  }
}
```

## Checklist sửa ESP32:

- [ ] Xóa dòng `openServo()` trong phần xử lý RFID
- [ ] Đảm bảo có `client.setCallback(callback)` trong setup()
- [ ] Đảm bảo có `client.loop()` trong loop()
- [ ] Đảm bảo subscribe topic `smart_parking/servo/command`
- [ ] Chỉ mở servo trong callback khi nhận lệnh từ backend
